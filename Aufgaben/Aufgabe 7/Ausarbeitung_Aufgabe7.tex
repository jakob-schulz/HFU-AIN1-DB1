\subsection{}
Erster Ansatz:
\begin{lstlisting}[style = sql]
Select distinct boat, count(*) as Anzahl_Passagiere
from titanic
where boat like '__' or boat like '_'
group by boat
having count(*) >= 25
\end{lstlisting}
Zweiter Ansatz:
\begin{lstlisting}[style = sql]
Select *
from 
(Select distinct boat, count(*) as Anzahl_Passagiere
from titanic
where boat like '__' or boat like '_'
group by boat) 
where Anzahl_Passagiere >=25
\end{lstlisting}
